# Cadastro de usuários
nome = input("Digite o nome do usuário: ")
endereco = input("Digite o endereço do usuário: ")
bairro = input("Digite o bairro do usuário: ")
cidade = input("Digite a cidade do usuário: ")
estado = input("Digite o estado do usuário: ")

# Limpa a tela
import os
os.system('cls' if os.name == 'nt' else 'clear')

# Cadastro de veículos
modelo = input("Digite o modelo do veículo: ")
marca = input("Digite a marca do veículo: ")
ano = int(input("Digite o ano do veículo: "))
combustivel = input("Digite o combustível do veículo: ")
tipo = input("Digite o tipo do veículo: ")
valor = float(input("Digite o valor do veículo: "))

# Cálculo do IPVA 2023
if ano <= 2009:
    ipva = 0.0
elif ano <= 2012:
    ipva = 0.01 * valor
elif ano <= 2016:
    ipva = 0.02 * valor
else:
    ipva = 0.03 * valor

# Cálculo do seguro
seguro = 0.03 * valor

# Imprime as informações do usuário e do veículo
print("Usuário:")
print("Nome:", nome)
print("Endereço:", endereco)
print("Bairro:", bairro)
print("Cidade:", cidade)
print("Estado:", estado)

print("\nVeículo:")
print("Modelo:", modelo)
print("Marca:", marca)
print("Ano:", ano)
print("Combustível:", combustivel)
print("Tipo:", tipo)
print("Valor:", valor)

print("\nImpostos:")
print("IPVA 2023:", ipva)
print("Seguro:", seguro)
